    <script>
        $(document).ready(function() {
            const form = $('#registrationForm');
            const previewSection = $('#previewSection');

            // Character counter
            $('#specialRequirements').on('keyup', function() {
                $('#charCount').text($(this).val().length);
            });

            // Focus event - highlight field
            $(form).find('input, select, textarea').on('focus', function() {
                $(this).parent().css('opacity', '1');
            });

            // Blur event - validate individual field
            $('#email').on('blur', function() {
                const email = $(this).val();
                const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                
                if (email === '') {
                    showError($(this), 'Email is required');
                } else if (!emailRegex.test(email)) {
                    showError($(this), 'Invalid email format');
                } else {
                    showSuccess($(this));
                }
            });

            $('#phone').on('blur', function() {
                const phone = $(this).val().replace(/\D/g, '');
                if (phone.length < 10) {
                    showError($(this), 'Phone must be at least 10 digits');
                } else {
                    showSuccess($(this));
                }
            });

            $('#firstName, #lastName').on('blur', function() {
                if ($(this).val().trim() === '') {
                    showError($(this), 'This field is required');
                } else if ($(this).val().trim().length < 2) {
                    showError($(this), 'Must be at least 2 characters');
                } else {
                    showSuccess($(this));
                }
            });

            // Show preview on input
            $(form).find('input, select, textarea').on('change keyup', function() {
                updatePreview();
            });

            // Form submission
            form.on('submit', function(e) {
                e.preventDefault();
                
                if (validateForm()) {
                    $('#successAlert').slideDown().delay(3000).slideUp();
                    form.trigger('reset');
                    previewSection.slideUp();
                    $('#charCount').text('0');
                } else {
                    $('#errorAlert').slideDown().delay(3000).slideUp();
                }
            });

            function validateForm() {
                let isValid = true;

                // Validate all required fields
                $(form).find('[required]').each(function() {
                    if ($(this).attr('type') === 'checkbox') {
                        if (!$(this).is(':checked')) {
                            showError($(this), 'This must be checked');
                            isValid = false;
                        }
                    } else if ($(this).val().trim() === '') {
                        showError($(this), 'This field is required');
                        isValid = false;
                    }
                });

                // Validate email
                const email = $('#email').val();
                const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
                if (email && !emailRegex.test(email)) {
                    showError($('#email'), 'Invalid email');
                    isValid = false;
                }

                // Validate phone
                const phone = $('#phone').val().replace(/\D/g, '');
                if (phone && phone.length < 10) {
                    showError($('#phone'), 'Invalid phone');
                    isValid = false;
                }

                return isValid;
            }

            function showError(element, message) {
                element.addClass('is-invalid').removeClass('is-valid');
                element.closest('.form-group').find('.error-message').text(message).show();
            }

            function showSuccess(element) {
                element.addClass('is-valid').removeClass('is-invalid');
                element.closest('.form-group').find('.error-message').hide();
            }

            function updatePreview() {
                const firstName = $('#firstName').val();
                const lastName = $('#lastName').val();
                const email = $('#email').val();
                const phone = $('#phone').val();
                const category = $('#eventType').val();
                const tshirt = $('#tshirtSize').val();
                const dietary = $('#dietaryPreference').val() || 'Not specified';

                if (firstName && lastName) {
                    $('#previewName').text(firstName + ' ' + lastName);
                    $('#previewEmail').text(email);
                    $('#previewPhone').text(phone);
                    $('#previewCategory').text(category);
                    $('#previewTshirt').text(tshirt);
                    $('#previewDietary').text(dietary);
                    previewSection.slideDown();
                } else {
                    previewSection.slideUp();
                }
            }

            // Set min date to today
            const today = new Date().toISOString().split('T')[0];
            $('#registrationDate').attr('min', today);
        });
    </script>
